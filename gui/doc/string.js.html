/**
 * @class newlines
 * @classdesc 							Convert newline characters to <br> tags
 *
 * @example
 * <div ng-bind="description | newlines"></div>
 *
 * @return {String} Reformatted text
 */
angular.module('toolkit-gui').filter('newlines', function () {
    return function(txt) {
        if (txt != null) {
            return txt.replace(/\n/g, '<br/>');
        } else {
            return '';
        }
    };
});
